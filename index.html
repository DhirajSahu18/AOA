            #include < stdio.h >

                void findMinMax(int arr[], int low, int high, int *min, int *max) {
                    int mid, min1, min2, max1, max2;
                    
                    // Base case: only one element
                    if (low == high) {
                        *min = *max = arr[low];
                        return;
                    }
                    
                    // Base case: two elements
                    if (high == low + 1) {
                        if (arr[low] > arr[high]) {
                            *max = arr[low];
                            *min = arr[high];
                        } else {
                            *max = arr[high];
                            *min = arr[low];
                        }
                        return;
                    }
                    
                    // Divide the array into two halves
                    mid = (low + high) / 2;
                    findMinMax(arr, low, mid, &min1, &max1);
                    findMinMax(arr, mid + 1, high, &min2, &max2);
                    
                    // Combine the results
                    if (min1 < min2) {
                        *min = min1;
                    } else {
                        *min = min2;
                    }
                    
                    if (max1 > max2) {
                        *max = max1;
                    } else {
                        *max = max2;
                    }
                }
                
                int main() {
                    int n, i;
                    int min, max;
                    int arr[100];
                    clrscr();
                    printf("Enter the size of the array: ");
                    scanf("%d", &n);
                    printf("Enter the elements of the array:\n");
                    for (i = 0; i  <  n; i++) {
                        scanf("%d", &arr[i]);
                    }
                    findMinMax(arr, 0, n - 1, &min, &max);
                    printf("Minimum element: %d\n", min);
                    printf("Maximum element: %d\n", max);
                    getch();
                    return 0;
                }

                int main() {
                    char code[MAX_CODE_LINES][100] = {
                        "int x = 5;",
                        "int y = 10;",
                        "int z = x + y;",
                        "int a = 2 * x;",
                        "int b = z + 5;",
                        "y = 20;",
                        "int c = z * y;"
                    };
                    int numLines = 7;
                
                    printf("Original Code:\n");
                    for (int i = 0; i < numLines; i++) {
                        printf("%s\n", code[i]);
                    }
                    
                    performDCE(code, numLines);
                    
                    printf("\nCode after Dead Code Elimination:\n");
                    for (int i = 0; i < numLines; i++) {
                        if (code[i][0] != '\0') {
                            printf("%s\n", code[i]);
                        }
                    }
                
                    return 0;
                }
                
                // Function to perform dead code elimination
                void performDCE(char code[][100], int numLines) {
                    bool isUsed[MAX_CODE_LINES] = {false};
                
                    for (int i = numLines - 1; i >= 0; i--) {
                        char *line = code[i];
                        char var;
                        int result = sscanf(line, "%*[^=]%c", &var);
                        if (result == 1 && var >= 'a' && var <= 'z') {
                            isUsed[i] = true;
                            for (int j = i + 1; j < numLines; j++) {
                                if (strstr(code[j], &var) != NULL) {
                                    isUsed[i] = false;
                                    break;
                                }
                            }
                        }
                    }
                
                    for (int i = 0; i < numLines; i++) {
                        if (!isUsed[i]) {
                            code[i][0] = '\0';
                        }
                    }
                }

                int lexical_analyzer(char *input_string) {
                    int current_state = 0;
                    char *ptr = input_string;
                
                    while (*ptr != '\0') {
                        switch (current_state) {
                            case 0:
                                if (*ptr == 'a')
                                    current_state = 1;
                                break;
                            case 1:
                                if (*ptr == 'b')
                                    current_state = 2;
                                else if (*ptr == 'a')
                                    current_state = 1;
                                else
                                    current_state = 0;
                                break;
                            case 2:
                                if (*ptr == 'b')
                                    current_state = 3;
                                else if (*ptr == 'a')
                                    current_state = 1;
                                else
                                    current_state = 0;
                                break;
                            case 3:
                                if (*ptr == 'a')
                                    current_state = 4;
                                else if (*ptr == 'b')
                                    current_state = 2;
                                else
                                    current_state = 0;
                                break;
                            case 4:
                                if (*ptr == 'a' || *ptr == 'b')
                                    current_state = 4;
                                else
                                    current_state = 0;
                                break;
                            default:
                                return 0;
                        }
                        if (current_state == 4)
                            return 1;
                        ptr++;
                    }
                    
                    return 0;
                }
                
                int main() {
                    char input_string[100];
                    printf("Enter a string to analyze: ");
                    gets(input_string);
                    
                    if (lexical_analyzer(input_string))
                        printf("String 'abba' is recognized\n");
                    else
                        printf("String 'abba' is not recognized\n");
                
                    getch();
                    return 0;
                }

            #define MAX_EXPR_LENGTH 100
            #define HASH_TABLE_SIZE 101
            
            typedef struct ExpressionNode {
                char expr[MAX_EXPR_LENGTH];
                struct ExpressionNode* next;
            } ExpressionNode;
            
            typedef struct {
                ExpressionNode* head;
            } HashTableNode;
            
            int hash(const char *str) {
                int hash = 0;
                while (*str) {
                    hash = (hash << 5) + *str++;
                }
                return hash % HASH_TABLE_SIZE;
            }
            
            void insert(HashTableNode* hashTable, const char* expr) {
                int index = hash(expr);
                ExpressionNode* newNode = (ExpressionNode*)malloc(sizeof(ExpressionNode));
                if (newNode == NULL) {
                    printf("Memory allocation failed!\n");
                    exit(1);
                }
                strcpy(newNode->expr, expr);
                newNode->next = hashTable[index].head;
                hashTable[index].head = newNode;
            }
            
            int contains(HashTableNode* hashTable, const char* expr) {
                int index = hash(expr);
                ExpressionNode* current = hashTable[index].head;
                while (current != NULL) {
                    if (strcmp(current->expr, expr) == 0) {
                        return 1; // Found
                    }
                    current = current->next;
                }
                return 0; // Not found
            }
            
            int main() {
                HashTableNode hashTable[HASH_TABLE_SIZE];
                int numExpr;
                int i;
                clrscr();
                printf("Enter the number of expressions: ");
                scanf("%d", &numExpr);
            
            
                for (i = 0; i < HASH_TABLE_SIZE; i++) {
                    hashTable[i].head = NULL;
                }
            
                // Input expressions
                printf("Enter the expressions:\n");
                for (i = 0; i < numExpr; i++) {
                    char expr[MAX_EXPR_LENGTH];
                    printf("Expression %d: ", i + 1);
                    scanf("%s", expr);
            
                    if (!contains(hashTable, expr)) {
                        insert(hashTable, expr);
                    }
                }
            
                // Output optimized expressions
                printf("\nOptimized expressions:\n");
                for (i = 0; i < HASH_TABLE_SIZE; i++) {
                    ExpressionNode* current = hashTable[i].head;
                    while (current != NULL) {
                        printf("%s\n", current->expr);
                        current = current->next;
                    }
                }
            
                // Free memory
                for (i = 0; i < HASH_TABLE_SIZE; i++) {
                    ExpressionNode* current = hashTable[i].head;

            void floyd_warshall(int graph[][100], int V) {
                int dist[100][100], i, j, k;
            
                // Initialize the distance matrix with the adjacency matrix
                for (i = 0; i < V; i++) {
                for (j = 0; j < V; j++) {
                    dist[i][j] = graph[i][j];
                }
                }
            
                // Calculate the shortest path for each pair of vertices
                for (k = 0; k < V; k++) {
                for (i = 0; i < V; i++) {
                    for (j = 0; j < V; j++) {
                    if (dist[i][k] + dist[k][j] < dist[i][j]) {
                        dist[i][j] = dist[i][k] + dist[k][j];
                    }
                    }
                }
                }
            
                // Print the shortest distance matrix
                printf("Shortest distance matrix:\n");
                for (i = 0; i < V; i++) {
                for (j = 0; j < V; j++) {
                    if (dist[i][j] == INF) {
                    printf("%7s", "INF");
                    } else {
                    printf("%7d", dist[i][j]);
                    }
                }
                printf("\n");
                }
            }
            
            int main() {
                int V, i, j;
                int graph[100][100];
                clrscr();
            
                // Get the number of vertices from the user
                printf("Enter the number of vertices: ");
                scanf("%d", &V);
            
                // Get the adjacency matrix from the user
                printf("Enter the adjacency matrix:\n");
                for (i = 0; i < V; i++) {
                for (j = 0; j < V; j++) {
                    scanf("%d", &graph[i][j]);
                    if (graph[i][j] == INF) {
                    graph[i][j] = INF;
                    }
                }
                }
            
                // Run the Floyd-Warshall algorithm
                floyd_warshall(graph, V);
                getch();
                return 0;
            }
